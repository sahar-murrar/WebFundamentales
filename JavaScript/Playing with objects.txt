3.How would you print/log the name and age of each user using a for loop?
var users = [
    {
        name: "Michael",
        age: 37
    },
    {
        name: "John",
        age: 30
    },
    {
        name: "David",
        age: 27
    }
];

for(var i=0; i<users.length; i++){
    console.log(users[i].name, "-", users[i].age);
}
